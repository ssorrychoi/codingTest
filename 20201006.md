function solution(n) {
    let answer = 2;

            for(let j=2;j<=parseInt(result[i]/2);j++){
                // console.log(`${result[i]},${j}`);
                if(result[i]%j==0){
                    cnt++;
                    break;
                }
            }
            if(cnt==0){
                answer++;
            }
            cnt=0;
        }
    }


    return answer;
}
